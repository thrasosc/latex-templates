% Redefine the existing theorem environments to be framed
\let\olddefinition\definition
\let\endolddefinition\enddefinition
\renewenvironment{definition}
  {\begin{framed}\olddefinition}
  {\endolddefinition\end{framed}}

\let\oldexample\example
\let\endoldexample\endexample
\renewenvironment{example}
  {\begin{framed}\oldexample}
  {\endoldexample\end{framed}}

\let\oldtheorem\theorem
\let\endoldtheorem\endtheorem
\renewenvironment{theorem}
  {\begin{framed}\oldtheorem}
  {\endoldtheorem\end{framed}}
